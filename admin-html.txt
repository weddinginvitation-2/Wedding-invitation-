// ════════════ JSONP ════════════
function callScriptOnce(params, timeoutMs) {
  return new Promise((resolve, reject) => {
    const cb = 'cb_' + Date.now() + '_' + Math.floor(Math.random()*1e6);
    const script = document.createElement('script');
    const timer = setTimeout(() => { cleanup(); reject(new Error('timeout')); }, timeoutMs || 30000);
    function cleanup(){ clearTimeout(timer); try{delete window[cb];}catch(e){window[cb]=undefined;} if(script.parentNode) script.parentNode.removeChild(script); }
    window[cb] = (data) => { cleanup(); resolve(data); };
    const qs = new URLSearchParams(Object.assign({}, params, {callback:cb})).toString();
    script.src = SCRIPT_URL + '?' + qs + '&_=' + Date.now();
    script.onerror = () => { cleanup(); reject(new Error('network')); };
    document.head.appendChild(script);
  });
}
async function callScript(params, attempts) {
  attempts = attempts || 3; let lastErr;
  for (let i = 0; i < attempts; i++) {
    try { return await callScriptOnce(params, 30000); }
    catch (e) { lastErr = e; await new Promise(r => setTimeout(r, 1500)); }
  }
  throw lastErr;
}

function doLogin() {

  if (pwd === ADMIN_PASSWORD) {

  }
}

// ════════════ حالة التسجيل ════════════
function renderRegState() {
  const st = document.getElementById('reg-state');
  const btn = document.getElementById('reg-toggle-btn');
  st.textContent = regOpen ? '🟢 مفتوح' : '🔴 مغلق';
  st.className = 'reg-state ' + (regOpen ? 'open' : 'closed');
  btn.style.display = 'inline-block';
  btn.textContent = regOpen ? 'إغلاق التسجيل' : 'فتح التسجيل';
}
async function toggleReg() {
  const btn = document.getElementById('reg-toggle-btn');
  btn.disabled = true;
  try {
    const res = await callScript({ action:'setreg', key:ADMIN_PASSWORD, open:String(!regOpen) });
    if (res.status === 'success') { regOpen = res.registrationOpen; renderRegState(); }
    else alert('⚠️ ' + (res.message || 'تعذّر تغيير الحالة'));
  } catch(e) { alert('⚠️ تعذّر الاتصال بالخادم'); }
  btn.disabled = false;
}

// ════════════ LOAD ════════════
async function loadAllGuests() {
  try {
    const data = await callScript({ action:'getguests' });
    allGuests = (data && data.guests) || [];
    if (typeof data.registrationOpen === 'boolean') { regOpen = data.registrationOpen; renderRegState(); }
    renderGuests();

      '<tr><td colspan="6"><div class="empty-state"><div class="emoji">⚠️</div>تعذّر الاتصال بالخادم</div></td></tr>';
  }
}

      <tr>
        <td>${i+1}</td>
        <td><strong>${esc(g.name)}</strong></td>
        <td class="code-cell">${esc(g.code)}</td>
        <td style="direction:ltr;text-align:right">${esc(g.phone)||'—'}</td>
        <td><span class="badge ${g.attended?'badge-green':'badge-gray'}">${g.attended?('✅ '+(esc(g.attendTime)||'حضر')):'⏳ لم يحضر'}</span></td>
        <td style="white-space:nowrap">
          <button class="btn btn-outline btn-sm" onclick="showGuestCard('${esc(g.code)}')" style="margin-left:4px">البطاقة</button>
          ${!g.attended ? `<button class="btn btn-green btn-sm" onclick="checkInByCode('${esc(g.code)}')">دخول</button>` : ''}
        </td>
      </tr>`).join('');
  }

  const attended = allGuests.filter(g=>g.attended).sort((a,b)=>String(b.attendTime||'').localeCompare(String(a.attendTime||'')));
  document.getElementById('attendance-list').innerHTML = attended.length
    ? attended.map(g => `<div class="log-entry"><div class="log-time">${esc(g.attendTime)||'—'}</div><div class="log-name"><strong>${esc(g.name)}</strong> <span class="code-cell" style="font-size:12px">${esc(g.code)}</span></div><span class="badge badge-green">✅ حضر</span></div>`).join('')
    : '<div class="empty-state"><div class="emoji">👋</div>لا يوجد حضور مسجل بعد</div>';

  const recent = attended.slice(0,6);
  document.getElementById('recent-arrivals').innerHTML = recent.length
    ? recent.map(g => `<tr><td><strong>${esc(g.name)}</strong></td><td class="code-cell">${esc(g.code)}</td><td>${esc(g.attendTime)||'—'}</td></tr>`).join('')
    : '<tr><td colspan="3"><div class="empty-state"><div class="emoji">🕐</div>لا يوجد حضور بعد</div></td></tr>';
}

function filterGuests() {
  const q = document.getElementById('guest-search').value.toLowerCase();
  const status = document.getElementById('filter-status').value;
  const filtered = allGuests.filter(g => {
    const mq = !q || String(g.name).toLowerCase().includes(q) || String(g.code).toLowerCase().includes(q);
    const ms = status==='all' || (status==='attended'&&g.attended) || (status==='pending'&&!g.attended);
    return mq && ms;
  });
  renderGuests(filtered);
}

function updateStats() {
  const total = allGuests.length;
  const attended = allGuests.filter(g=>g.attended).length;
  document.getElementById('stat-total').textContent = total;
  document.getElementById('stat-attended').textContent = attended;
  document.getElementById('stat-pending').textContent = total-attended;
  document.getElementById('stat-pct').textContent = (total?Math.round(attended/total*100):0)+'%';
}

// ════════════ البطاقة المؤطّرة (QR داخل الإطار) ════════════
function makeQrCanvas(text, size) {
  const holder = document.createElement('div');
  holder.style.cssText = 'position:fixed;left:-9999px;top:0';
  document.body.appendChild(holder);
  new QRCode(holder, { text:text, width:size, height:size,
    colorDark:'#5E1520', colorLight:'#ffffff', correctLevel:QRCode.CorrectLevel.H });
  const cv = holder.querySelector('canvas');
  const out = document.createElement('canvas');
  out.width = size; out.height = size;
  out.getContext('2d').drawImage(cv, 0, 0, size, size);
  holder.remove();
  return out;
}
function roundRect(ctx,x,y,w,h,r){ctx.beginPath();ctx.moveTo(x+r,y);ctx.arcTo(x+w,y,x+w,y+h,r);ctx.arcTo(x+w,y+h,x,y+h,r);ctx.arcTo(x,y+h,x,y,r);ctx.arcTo(x,y,x+w,y,r);ctx.closePath();}

async function buildCard(name, code, targetId) {
  const qrCv = makeQrCanvas('WEDDING:' + code + ':' + name, FRAME.qr.size);
  const canvas = document.createElement('canvas');
  canvas.width = FRAME.width; canvas.height = FRAME.height;
  const ctx = canvas.getContext('2d');
  try { await document.fonts.ready; } catch(e){}

  const drawTexts = () => {
    ctx.textAlign = 'center';
    ctx.fillStyle = FRAME.name.color;
    ctx.font = '700 ' + FRAME.name.size + 'px "Aref Ruqaa", "Amiri", serif';
    ctx.fillText(name, FRAME.width/2, FRAME.name.y);
    ctx.fillStyle = FRAME.code.color;
    ctx.font = '700 ' + FRAME.code.size + 'px monospace';
    ctx.fillText(code, FRAME.width/2, FRAME.code.y);
  };
  const drawQr = () => {
    const q = FRAME.qr, p = q.pad;
    roundRect(ctx, q.x-p, q.y-p, q.size+p*2, q.size+p*2, 20);
    ctx.fillStyle = '#fff'; ctx.fill();
    ctx.drawImage(qrCv, q.x, q.y, q.size, q.size);
  };
  const show = () => {
    const t = document.getElementById(targetId);
    t.innerHTML = ''; t.appendChild(canvas);
  };

  const frameImg = new Image();
  frameImg.onload = () => { ctx.drawImage(frameImg,0,0,FRAME.width,FRAME.height); drawQr(); drawTexts(); show(); };
  frameImg.onerror = () => {
    ctx.fillStyle = '#FAF6FD'; ctx.fillRect(0,0,FRAME.width,FRAME.height);
    ctx.strokeStyle = '#D4572B'; ctx.lineWidth = 10; ctx.strokeRect(40,40,FRAME.width-80,FRAME.height-80);
    ctx.lineWidth = 2; ctx.strokeRect(60,60,FRAME.width-120,FRAME.height-120);
    ctx.textAlign = 'center'; ctx.fillStyle = '#5E1520';
    ctx.font = '700 64px "Aref Ruqaa", serif';
    ctx.fillText('حفل زفاف محمود & إسراء', FRAME.width/2, 200);
    ctx.fillStyle = '#8E7A92'; ctx.font = '32px "Amiri", serif';
    ctx.fillText('الاثنين ٢٧ يوليو ٢٠٢٦ — قاعة فندق البوليڤارد', FRAME.width/2, 270);
    drawQr(); drawTexts(); show();
  };
  frameImg.src = FRAME.url + '?v=' + Date.now();
}

function showGuestCard(code) {
  const g = allGuests.find(x => String(x.code)===String(code));
  if (!g) return;
  document.getElementById('modal-guest-name').textContent = g.name;
  document.getElementById('modal-code').textContent = g.code;
  document.getElementById('modal-card-preview').innerHTML = '<div class="empty-state">⏳</div>';
  buildCard(g.name, g.code, 'modal-card-preview');
  const btn = document.getElementById('modal-checkin-btn');
  if (g.attended) { btn.textContent='✅ حضر مسبقاً'; btn.disabled=true; btn.onclick=null; }
  else { btn.textContent='✅ تسجيل الحضور'; btn.disabled=false; btn.onclick=()=>{ checkInByCode(g.code); closeModal(); }; }
  document.getElementById('qr-modal').classList.add('open');
}
function closeModal(){ document.getElementById('qr-modal').classList.remove('open'); }

// ════════════ CHECK-IN ════════════
async function checkInByCode(code) {
  try {
    const res = await callScript({ action:'checkin', code:code });
    if (res.status==='success') {
      const g = allGuests.find(x=>String(x.code)===String(code));
      if (g){ g.attended=true; g.attendTime=res.time; }
      renderGuests(); updateStats();
      alert('✅ تم تسجيل حضور '+(res.name||code));
    } else if (res.status==='duplicate') {
      alert('⚠️ '+(res.name||code)+' سجّل دخوله مسبقاً ('+(res.attendTime||'')+')');
      loadAllGuests();
    } else {
      alert('❌ '+(res.message||'الكود غير موجود'));
    }
  } catch(e) { alert('⚠️ تعذّر الاتصال بالخادم'); }
}

// ════════════ ADD GUEST ════════════
async function addGuestManually() {
  const name = document.getElementById('add-name').value.trim();
  const phone = document.getElementById('add-phone').value.trim();
  const msgEl = document.getElementById('add-msg');
  const btn = document.getElementById('add-btn');
  if (!name) { msgEl.textContent='⚠️ الرجاء إدخال الاسم'; return; }
  btn.disabled=true; msgEl.textContent='جاري الإضافة...';
  try {
    const res = await callScript({ action:'rsvp', name:name, phone:phone });
    if (res.status==='success' || res.status==='exists') {
      const code = res.code;
      document.getElementById('new-card-name').textContent = res.name || name;
      document.getElementById('new-card-code').textContent = code;
      buildCard(res.name || name, code, 'new-card-preview');
      document.getElementById('new-guest-card').style.display='block';
      msgEl.textContent = res.status==='exists' ? ('ℹ️ مسجّل مسبقاً — كوده: '+code) : ('✅ تمت إضافة '+(res.name||name));
      document.getElementById('add-name').value=''; document.getElementById('add-phone').value='';
      loadAllGuests();
    } else if (res.status==='closed') {
      msgEl.textContent = '⚠️ التسجيل مغلق — افتحه من الشريط بالأعلى ثم أعد المحاولة';
    } else {
      msgEl.textContent = '⚠️ '+(res.message||'خطأ');
    }
  } catch(e) { msgEl.textContent='⚠️ تعذّر الاتصال بالخادم'; }
  btn.disabled=false;
}

}
